Love Coefficient Computation
export function computeLoveCoefficient(state: CoherenceState): LoveCoefficient {
  const { F, L, H } = state;
  const clamp = (v: number) => Math.max(0, Math.min(1, v));
  const f = clamp(F);
  const l = clamp(L);
  const h = clamp(H);
  const L_final = l * (0.5 + 0.3 * f + 0.2 * h);
  return Math.round(L_final * 10000) / 10000;
}


MZS Prophetic Lock
export function enforceMZSLock(love: LoveCoefficient, strict = true) {
  if (!strict) {
    return { locked: false, reason: 'MZS lock disabled (non-strict mode)' };
  }
  const locked = love >= 1.0;
  if (locked) {
    return { locked: true, reason: 'MZS Prophetic Lock satisfied (ℒ >= 1.0)' };
  } else {
    return { locked: false, reason: `MZS Prophetic Lock failed (ℒ=${love} < 1.0)` };
  }
}


Coherent Action Mandate Generation
export function generateCoherentActionMandate(
  state: CoherenceState,
  summaryHint?: string,
  options?: { requireProof?: boolean; mzsStrict?: boolean }
): { mandate: CoherentActionMandate; audit: CovenantAuditRecord } {
  const now = new Date().toISOString();
  const love = computeLoveCoefficient(state);
  const mzs = enforceMZSLock(love, options?.mzsStrict ?? true);
  const summary = summaryHint || generateSmallActionFromState(state);
  
  const mandate: CoherentActionMandate = {
    id: uuidv4(),
    summary,
    rationale: `Generated from fusedState (fusedScore=${state.fusedScore}, ℒ=${love})`,
    generatedAt: now,
    requiredProofOfAction: options?.requireProof ?? true,
    mzsLocked: mzs.locked,
    suggestedEvidence: ['photo', 'log', 'timestamp'],
    metadata: { loveCoefficient: love, mzsReason: mzs.reason },
  };
  
  const audit: CovenantAuditRecord = {
    id: uuidv4(),
    mandate,
    coherenceStateAtGeneration: state,
    loveCoefficientAtGeneration: love,
    outcome: 'pending',
    proof: null,
    notes: mzs.locked ? 'MZS satisfied' : 'MZS failed',
    createdAt: now,
  };
  
  return { mandate, audit };
}

import {
  CovenantAuditRecord,
  ProofOfAction,
  CoherenceState,
} from './types';
import { generateCoherentActionMandate, computeLoveCoefficient } from './engine';


export type CovenantStore = {
  saveAudit: (record: CovenantAuditRecord) => Promise<void>;
  getAuditById: (id: string) => Promise<CovenantAuditRecord | null>;
  attachProof: (id: string, proof: ProofOfAction) => Promise<void>;
};


export async function createMandateAndSave(
  store: CovenantStore,
  state: CoherenceState,
  summaryHint?: string
): Promise<CovenantAuditRecord> {
  const { audit } = generateCoherentActionMandate(state, summaryHint, {
    requireProof: true,
    mzsStrict: true,
  });
  await store.saveAudit(audit);
  return audit;
}


export async function verifyProofOfAction(
  store: CovenantStore,
  mandateId: string,
  proof: ProofOfAction
): Promise<{ ok: boolean; reason?: string }> {
  const audit = await store.getAuditById(mandateId);
  if (!audit) return { ok: false, reason: 'audit not found' };
  if (!audit.mandate.requiredProofOfAction) {
    return { ok: false, reason: 'not required' };
  }
  if (proof.mandateId !== mandateId) {
    return { ok: false, reason: 'id mismatch' };
  }
  if (!proof.evidenceUri) {
    return { ok: false, reason: 'missing evidence' };
  }
  await store.attachProof(mandateId, proof);
  return { ok: true };
}


export function loveForState(state: CoherenceState) {
  return computeLoveCoefficient(state);
}
